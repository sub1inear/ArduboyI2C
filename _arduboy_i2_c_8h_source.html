515// counts the number of edges on the SDA and SCL lines to determine if the cable is flipped
516// relies on I2C::handshake() sending 0b00000000 so that the SCL line is toggled and the SDA line is held low
517bool checkCableFlippedCore(bool disconnectFlip) {
518 uint8_t prev = I2C_PIN;
519 uint8_t sdaEdges = 0;
520 uint8_t sclEdges = 0;
521
522 for (uint8_t i = 0; i < 128; i++) {
523 uint8_t cur = I2C_PIN;
524 // calculates the change between the current and previous pin states
525 uint8_t diff = cur ^ prev;
526
527 // if the SDA line changed, increment the SDA edge counter
528 if (diff & _BV(I2C_SDA_BIT)) { sdaEdges++; }
529 // if the SCL line changed, increment the SCL edge counter
530 if (diff & _BV(I2C_SCL_BIT)) { sclEdges++; }
531
532 prev = cur;
533 // half-period delay otherwise too fast
534 _delay_us(1000000.0 / I2C_FREQUENCY / 2.0);
535 }
536
537 // if the total number of edges is very low (<= 2 to account for noise), the cable is disconnected or flipped
538 if (sdaEdges + sclEdges <= 2) {
539 // returns flipped (true) if disconnect is a flip, otherwise returns not flipped (false)
540 return disconnectFlip;
541 }
542 // otherwise, if the number of SDA edges is greater than the number of SCL edges, the cable is flipped
543 return sdaEdges > sclEdges;
544}

570void I2C::begin() {
571 // power up TWI
572 // power_twi_disable() is called in arduboy.boot() (inside arduboy.begin())
573 power_twi_enable();
574
575 // enable TWI, interrupts, ACKing
576 TWCR = _BV(TWEN) | _BV(TWIE) | _BV(TWEA);
577 // set our clock frequency
578 TWBR = (F_CPU / I2C_FREQUENCY - 16) / 2;
579
580 // enable software pullups (needed for Arduboy FX-C)
581 I2C_DDR &= ~(_BV(I2C_SDA_BIT) | _BV(I2C_SCL_BIT));
582 I2C_PORT |= _BV(I2C_SDA_BIT) | _BV(I2C_SCL_BIT);
583}
584
585void I2C::end() {
586 // disable TWI
587 TWCR = 0;
588
589 // power down TWI
590 power_twi_disable();
591
592 // disable software pullups (needed for Arduboy FX-C)
593 // no need to change DDR
594 I2C_PORT &= ~(_BV(I2C_SDA_BIT) | _BV(I2C_SCL_BIT));
595}

620void I2C::read(uint8_t address, void *buffer, uint8_t size) {
621 i2c_detail::waitActive();
622
623 // ISR: TWCR = i2c_detail::bufferIdx < i2c_detail::bufferSize ? REPLY_ACK : REPLY_NACK;
624 // so we must set the buffer size to size - 1 so that the last byte is NACKed (I2C standard)
625 i2c_detail::startReadWrite(address, TW_READ, size - 1);
626 i2c_detail::data.readBuffer = (uint8_t *)buffer;
627
628 i2c_detail::sendStart();
629
630 i2c_detail::waitActive();
631}

659void I2C::checkCableFlipped(void (*startFunction)(), void (*loopFunction)()) {
660 // disable TWI
661 TWCR = 0;
662
663 // check if cable is flipped
664 // pass false to indicate a disconnect is not a flip
665 // so it will return false and we will continue
666 if (i2c_detail::checkCableFlippedCore(false)) {
667 if (startFunction) {
668 startFunction();
669 }
670 // wait for the cable to be flipped back
671 // debounce the cable flip detection by ensuring 64 counts of success
672 // putting the cable back it can generate noise
673 for (uint8_t i = 0; i < 64; i++) {
674 // pass true to indicate a disconnect is a flip
675 // so we will wait while the cable is disconnected
676 if (i2c_detail::checkCableFlippedCore(true)) {
677 i = 0;
678 }
679 if (loopFunction) {
680 loopFunction();
681 }
682 }
683 }
684
685 // re-enable TWI
686 TWCR = _BV(TWEN) | _BV(TWIE) | _BV(TWEA);
687}

689bool I2C::handshake(void (*startFunction)(), void (*loopFunction)()) {
690 // check if the bus is free (SDA and SCL are high for 128 half-periods)
691 // if it is not, another controller (master) has gotten here before us
692 for (uint8_t i = 0; i < 128; i++) {
693 if ((I2C_PIN & (_BV(I2C_SDA_BIT) | _BV(I2C_SCL_BIT))) !=
694 (_BV(I2C_SDA_BIT) | _BV(I2C_SCL_BIT))) {
695 // set our address to the target address
696 I2C::setAddress(I2C_TARGET_ADDRESS);
697 // wait for twice the time it takes to send 18 bits
698 // (7 address bits + 1 RW + 1 ACK + 8 data bits + 1 ACK) to ensure we ACK the controller
699 // might have come across a controller that just started sending data,
700 // so we will NACK the first transmission and the ACK the second transmission
701 // so we need to wait for two transmissions
702 _delay_us((1000000.0 / I2C_FREQUENCY * 18.0 + 25.0) * 2.0);
703 // return `false` to indicate that this device is the target (slave)
704 return false;
705 }
706 // half-period delay otherwise too fast
707 _delay_us(1000000.0 / I2C_FREQUENCY / 2.0);
708 }
709 uint8_t zeros = 0b00000000;
710 if (startFunction) {
711 startFunction();
712 }
713 do {
714 // send all zeros to the target address to help with I2C::checkCableFlipped()
715 I2C::write(I2C_TARGET_ADDRESS, zeros, true);
716 if (loopFunction) {
717 loopFunction();
718 }
719 // repeat until the write is successful (no NACKs)
720 } while (I2C::getError() != I2C_ERROR_NONE);
721 // return `true` to indicate that this device is the controller (master)
722 return true;
723}
