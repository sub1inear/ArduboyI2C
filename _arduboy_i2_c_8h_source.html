566#if I2C_USE_MULTI_CONTROLLER
567bool checkBusBusy() {
568 uint8_t busyChecks = I2C_MULTI_CONTROLLER_BUSY_CHECKS;
569 while (busyChecks) {
570 if ((I2C_PIN & _BV(I2C_SDA_BIT)) && (I2C_PIN & _BV(I2C_SCL_BIT))) {
571 busyChecks--;
572 } else {
573 i2c_detail::data.error = TW_MT_ARB_LOST; // same as TW_MR_ARB_LOST
574 i2c_detail::data.active = false;
575 return true;
576 }
577 }
578 return false;
579}
580#endif // #if I2C_USE_MULTI_CONTROLLER

582#if I2C_USE_CHECK_CABLE_FLIPPED
583bool checkCableFlippedCore(bool disconnectFlip = false) {
584 // count frequency of each state of the SDA and SCL lines
585 uint8_t sdaHigh = 0;
586 uint8_t sclHigh = 0;
587 for (uint8_t i = 0; i < I2C_CHECK_CABLE_FLIPPED_CHECKS; i++) {
588 // probably should buffer the pin reads
589 // but saves an instruction not to
590 // and will only be off by a very small amount
591 if (I2C_PIN & _BV(I2C_SDA_BIT)) { sdaHigh++; }
592 if (I2C_PIN & _BV(I2C_SCL_BIT)) { sclHigh++; }
593 // half-period delay (otherwise way too fast to detect changes)
594 // double will be optimized away into constant cycle loop
595 _delay_us(1000000.0 / I2C_FREQUENCY / 2.0);
596 }
597 // if the cable disconnected (both lines high) for the entire check
598 // and disconnectFlip is true, return true (flipped)
599 // otherwise will return false in same case
600 // (sdaScore = halfChecks, sclScore = halfChecks, sdaScore < sclScore is false)
601 if (disconnectFlip &&
602 sdaHigh == I2C_CHECK_CABLE_FLIPPED_CHECKS &&
603 sclHigh == I2C_CHECK_CABLE_FLIPPED_CHECKS) {
604 return true;
605 }
606 // score each line based on its distance from half the checks being high
607 constexpr uint8_t halfChecks = I2C_CHECK_CABLE_FLIPPED_CHECKS / 2;
608 uint8_t sdaScore = (uint8_t)abs((int8_t)(sdaHigh - halfChecks));
609 uint8_t sclScore = (uint8_t)abs((int8_t)(sclHigh - halfChecks));
610 // less score means more like a clock
611 // so flipped if sdaScore is less than sclScore
612 return sdaScore < sclScore;
613}
614// optimizes for debounce in checkCableFlipped (only needs uint16_t)
615uint16_t millisShort() {
616 return (uint16_t)millis();
617}

620void startReadWrite(uint8_t address, bool readWrite) {
621 while (i2c_detail::data.active) {}
622
623 i2c_detail::data.error = TW_SUCCESS;
624 i2c_detail::data.slaRW = address << 1 | readWrite;
625 i2c_detail::data.bufferIdx = 0;

630void I2C::begin() {
631 power_twi_enable();
632 TWCR = _BV(TWEN) | _BV(TWIE) | _BV(TWEA);
633
634 TWBR = (F_CPU / I2C_FREQUENCY - 16) / 2;
635
636 // clear prescaler bits
637 TWSR = 0;
638
639#if I2C_USE_SOFTWARE_PULLUPS
640 // enable software pullups
641 I2C_DDR &= ~(_BV(I2C_SDA_BIT) | _BV(I2C_SCL_BIT));
642 I2C_PORT |= _BV(I2C_SDA_BIT) | _BV(I2C_SCL_BIT);
643#endif // #if I2C_USE_SOFTWARE_PULLUPS
644}

650void I2C::write(uint8_t address, const void *buffer, uint8_t size, bool wait) {
651 i2c_detail::startReadWrite(address, TW_WRITE);
652
653 memcpy(i2c_detail::data.twiBuffer, buffer, size);
654
655 i2c_detail::data.bufferSize = size;
656
657#if I2C_USE_MULTI_CONTROLLER
658 if (i2c_detail::checkBusBusy()) { return; }
659 // note: TOCTOU race condition here, but it's unavoidable
660 // no way to fix while prioritizing target (slave) interrupts so handshaking is solid
661 if (i2c_detail::data.active) {
662 i2c_detail::data.error = TW_MT_ARB_LOST; // same as TW_MR_ARB_LOST
663 return;
664 }
665#endif // #if I2C_USE_MULTI_CONTROLLER
666 i2c_detail::data.active = true;
667 TWCR = _BV(TWEN) | _BV(TWIE) | _BV(TWSTA) | _BV(TWINT);
668 if (wait) {
669 while (i2c_detail::data.active) {}
670 }
671}

673void I2C::read(uint8_t address, void *buffer, uint8_t size) {
674 i2c_detail::startReadWrite(address, TW_READ);
675
676 i2c_detail::data.rxBuffer = (uint8_t *)buffer;
677 i2c_detail::data.bufferSize = size - 1;
678
679#if I2C_USE_MULTI_CONTROLLER
680 if (i2c_detail::checkBusBusy()) { return; }
681 // note: TOCTOU race condition here, but it's unavoidable
682 // no way to fix while prioritizing target (slave) interrupts so handshaking is solid
683 if (i2c_detail::data.active) {
684 i2c_detail::data.error = TW_MT_ARB_LOST; // same as TW_MR_ARB_LOST
685 return;
686 }
687#endif // #if I2C_USE_MULTI_CONTROLLER
688 i2c_detail::data.active = true;
689 TWCR = _BV(TWEN) | _BV(TWIE) | _BV(TWSTA) | _BV(TWINT);
690 while (i2c_detail::data.active) {}
691}

718#if I2C_USE_CHECK_CABLE_FLIPPED
719void I2C::checkCableFlipped(void (*function)()) {
720 // wait to finish ongoing operations
721 while (i2c_detail::data.active) { }
722 TWCR = 0; // disable TWI
723
724 if (i2c_detail::checkCableFlippedCore()) {
725 // inform the user of the flipped cable
726 function();
727 // wait for cable to be flipped back
728 // debounce cable changes
729 uint16_t start = i2c_detail::millisShort();
730 while (true) {
731 if (i2c_detail::checkCableFlippedCore(true)) {
732 start = i2c_detail::millisShort();
733 } else if (i2c_detail::millisShort() - start >
734 I2C_CHECK_CABLE_FLIPPED_DEBOUNCE_TIMEOUT) {
735 break;
736 }
737 }
738 }
739
740 TWCR = _BV(TWEN) | _BV(TWIE) | _BV(TWINT); // re-enable TWI
741}
742#endif // #if I2C_USE_CHECK_CABLE_FLIPPED

744bool I2C::checkEmulator() {
745 // TWWC is set when TWDR is written to without TWINT being set
746 // not done in emulator
747 TWDR = 0;
748 return !(TWCR & _BV(TWWC));
749}

755#if I2C_USE_HANDSHAKE
756uint8_t I2C::handshake(uint8_t numPlayers) {
757 for (int8_t i = numPlayers - 1; i >= 0; ) {
758 uint8_t dummy;
759 uint8_t address = I2C::idToAddress(i);
760
761 I2C::read(address, dummy);
762
763 switch (I2C::getError()) {
764 case TW_MR_SLA_NACK:
765 I2C::onRequest(i2c_detail::handshakeOnRequest);
766 I2C::setAddress(address);
767
768 // handshakeState is the number of times the callback has been called.
769 // when the callback has been called i times, the final Arduboy has joined.
770 // cable flipped detection relies on clock detection,
771 // so we send 0b00000000 to have SDA never change
772 // while detecting it.
773
774#if I2C_USE_CHECK_CABLE_FLIPPED
775 dummy = 0b00000000;
776 while (i2c_detail::handshakeState < i) {
777 I2C::write(I2C_GENERAL_CALL, dummy, true);
778 }
779#else
780 while (i2c_detail::handshakeState < i) { }
781#endif // #if I2C_USE_CHECK_CABLE_FLIPPED
782
783 return i;
784 case TW_SUCCESS:
785 i--;
786 break;
787 }
788 }
789 return I2C_HANDSHAKE_FULL;
790}

792#endif // #if I2C_USE_HANDSHAKE

794ISR(TWI_vect, ISR_NAKED) {
795 asm volatile (
796R"(
797; --------------------- defines ----------------------- ;
798.equ TWPTR, 0xB9
799.equ TWCR, 3
800.equ TWSR, 0
801.equ TWDR, 2
802
803.equ TWIE, 0
804.equ TWEN, 2
805.equ TWWC, 3
806.equ TWSTO, 4
807.equ TWSTA, 5
808.equ TWEA, 6
809.equ TWINT, 7
810
811.equ REPLY_ACK, (1 << TWINT) | (1 << TWEN) | (1 << TWIE) | (1 << TWEA)
812.equ REPLY_NACK, (1 << TWINT) | (1 << TWEN) | (1 << TWIE)
813.equ STOP, (1 << TWINT) | (1 << TWEN) | (1 << TWIE) | (1 << TWSTO) | (1 << TWEA)
814
815; -------------------- registers ---------------------- ;
816; r18 - TWSR (never used after function call)
817; r19 - general use
818; r20 - REPLY_ACK
819; r21 - REPLY_NACK
820; r26 (X) - general use
821; r27 (X) - general use
822; r28 (Y) - data pointer
823; r29 (Y) - data pointer
824; r30 (Z) - TW register pointer
825; r31 (Z) - TW register pointer
826; --------------------- prologue ---------------------- ;
827push r18
828in r18, __SREG__
829push r18
830; save and restore call-clobbered registers
831; target (slave) could call function pointer
832push r19
833push r20
834push r21
835push r22
836push r23
837push r24
838push r25
839push r26
840push r27
841push r28
842push r29
843push r30
844push r31
845; save and restore tmp and zero registers (could be used in function calls)
846push __tmp_reg__
847push __zero_reg__
848clr __zero_reg__
849; ----------------------------------------------------- ;
850; set up Y pointer (data)
851ldi r28, lo8(%[data])
852ldi r29, hi8(%[data])
853
854; set up Z pointer (TW registers)
855ldi r30, TWPTR
856clr r31
857
858; set up r20 and r21 (REPLY_ACK and REPLY_NACK)
859ldi r20, REPLY_ACK
860ldi r21, REPLY_NACK
861
862; switch (TWSR)
863ldd r18, Z + TWSR ; no mask needed because prescaler bits are cleared
864
865cpi r18, 0x08
866breq TW_START
867
868; MT_MR
869cpi r18, 0x18
870breq TW_MT_SLA_ACK
871cpi r18, 0x28
872breq TW_MT_DATA_ACK
873cpi r18, 0x38
874breq TW_MT_ARB_LOST ; same as TW_MR_ARB_LOST
875cpi r18, 0x40
876breq TW_MR_SLA_ACK
877cpi r18, 0x50
878breq TW_MR_DATA_ACK
879cpi r18, 0x58
880breq TW_MR_DATA_NACK
881
882; 64 instruction limit on branches
883rjmp SR_ST
884
885TW_START:
886 ; TWDR = i2c_detail::data.slaRW;
887 ldd r26, Y + %[slaRW]
888 std Z + TWDR, r26
889 ; TWCR = REPLY_NACK;
890 std Z + TWCR, r21
891 ; return;
892 rjmp pop_reti
893
894TW_MT_SLA_ACK:
895TW_MT_DATA_ACK:
896 ; if (i2c_detail::data.bufferIdx >= i2c_detail::data.bufferSize) { stop(); return; }
897 ldd r26, Y + %[bufferIdx]
898 ldd r27, Y + %[bufferSize]
899 cp r26, r27
900
901 brlo 1f ; 64 instruction limit on branches
902 rjmp stop_reti
903 1:
904
905 ; TWDR = i2c_detail::data.twiBuffer[i2c_detail::data.bufferIdx++];
906 call get_buffer_addr
907 ld r26, X
908 std Z + TWDR, r26
909
910 ; TWCR = REPLY_NACK;
911 std Z + TWCR, r21
912 ; return;
913 rjmp pop_reti
914
915TW_MT_ARB_LOST:
916 ; TWCR = REPLY_ACK;
917 std Z + TWCR, r20
918 ; i2c_detail::data.error = TW_MT_ARB_LOST;
919 std Y + %[error], r18
920 ; active = false;
921 ; return;
922 rjmp active_false_reti
923; ----------------------------------------------------- ;
924
925TW_MR_DATA_NACK:
926TW_MR_DATA_ACK:
927 ; i2c_detail::data.rxBuffer[i2c_detail::data.bufferIdx++] = TWDR;
928 ldd r19, Y + %[bufferIdx]
929 ldd r26, Y + %[rxBuffer]
930 ldd r27, Y + %[rxBuffer] + 1
931
932 add r26, r19
933 adc r27, __zero_reg__
934
935 inc r19
936 std Y + %[bufferIdx], r19
937
938 ldd r19, Z + TWDR
939 st X, r19
940
941 ; if (TWSR == TW_MR_DATA_NACK) { stop(); return; }
942 ; r18 holds TWSR
943 cpi r18, 0x58
944 brne 1f ; 64 instruction limit on branches
945 rjmp stop_reti
946 1:
947; ------------------ fallthrough ---------------------- ;
948TW_MR_SLA_ACK:
949 ; if (i2c_detail::data.bufferIdx < i2c_detail::data.bufferSize) {
950 ; TWCR = REPLY_ACK;
951 ; } else {
952 ; TWCR = REPLY_NACK;
953 ; }
954 ; return;
955
956 ; r20 and r21 may be clobbered, do not use
957
958 ldd r26, Y + %[bufferIdx]
959 ldd r27, Y + %[bufferSize]
960 cp r26, r27
961 ldi r26, REPLY_ACK
962 brlo 1f
963 ldi r26, REPLY_NACK
964 1:
965 std Z + TWCR, r26
966 rjmp pop_reti
967; ----------------------------------------------------- ;
968SR_ST:
969cpi r18, 0x60
970breq TW_SR_SLA_ACK
971cpi r18, 0x68
972breq TW_SR_ARB_LOST_SLA_ACK
973cpi r18, 0x70
974breq TW_SR_GCALL_ACK
975cpi r18, 0x78
976breq TW_SR_ARB_LOST_GCALL_ACK
977cpi r18, 0x80
978breq TW_SR_DATA_ACK
979cpi r18, 0x90
980breq TW_SR_GCALL_DATA_ACK
981cpi r18, 0xA0
982breq TW_SR_STOP
983cpi r18, 0xA8
984breq TW_ST_SLA_ACK
985cpi r18, 0xB0
986breq TW_ST_ARB_LOST_SLA_ACK
987cpi r18, 0xB8
988breq TW_ST_DATA_ACK
989cpi r18, 0xC0
990breq TW_ST_DATA_NACK
991cpi r18, 0xC8
992breq TW_ST_LAST_DATA
993
994rjmp default
995
996TW_SR_SLA_ACK:
997TW_SR_ARB_LOST_SLA_ACK:
998TW_SR_GCALL_ACK:
